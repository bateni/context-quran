\startbuffer[\q:17:87]

\stopbuffer%

\stopbuffer%
\startbuffer[\q:17:92]

\stopbuffer%
\startbuffer[\q:17:93]

\stopbuffer%
\startbuffer[\q:17:96]

\stopbuffer%
